def annualized_return(r, periods_per_year):
    """
    Computes annualized return on a time series of asset returns
    """
    compounded_growth = (r + 1).prod()
    n_periods = r.shape[0]
    return compounded_growth ** (periods_per_year / n_periods) - 1
